    $ # for hubot-auth

    $ # for hubot-backlog-activity
    $ export HUBOT_BACKLOG_ACTIVITY_API_KEY='...'

    $ export HUBOT_BACKLOG_ACTIVITY_SPACE_ID='...'

    $ export HUBOT_BACKLOG_ACTIVITY_USE_SLACK='1'
